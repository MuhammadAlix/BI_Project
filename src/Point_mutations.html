fn main() {
    let a = "atcggtacga";
    let b = "ataggaaaga";
    let mut c = 0; // Define `c` as mutable

    let a_chars: Vec = a.chars().collect();
    let b_chars: Vec = b.chars().collect();

    let a_len = a_chars.len();
    let b_len = b_chars.len();

    if a_len != b_len {
        println!("Strings are of different lengths");
        return;
    }

    for i in 0..a_len {
        if a_chars[i] != b_chars[i] {
            c += 1;
        }
    }

    println!("{}", c);
}
